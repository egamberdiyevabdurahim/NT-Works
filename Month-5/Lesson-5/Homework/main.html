Aggregate Function, Filtering in SQL


1) Fist(hamma userlarni first_name va last_name larini olib bering):

SELECT first_name, last_name FROM users GROUP BY first_name, last_name;

2) Second(first_name ni ichida l harfi borlarini first_name, last_name, username larini olib bering):

SELECT first_name, last_name, username FROM users WHERE first_name LIKE '%l%' GROUP BY username, first_name, last_name;

3) Third(Faqatgina xitoy davlatida tug'ilgan odamlarni olib bering va ular takrorlanmasin):

SELECT DISTINCT * FROM users WHERE country = 'China';

4) Fourth(Har bitta davlatdan nechtadan foydalanuvchi borlini chiqarib bering):

SELECT country, COUNT(*) FROM users GROUP BY country;

5) Fifth(sizda 4 ta davlar tuple ko'rinishida turibdi, usha davlatlarda tug'ilgan odamlarni olib being):

SELECT * FROM users WHERE country IN ('Uzbekistan', 'Syria', 'Pakistan');

6) Sixth(Telefon raqami oxiri 22 bilan tugagan va emaili bor bo'lgan odamlarni username, phone va emaillarini olib bering):

SELECT username, phone_number, email FROM users WHERE phone_number LIKE '%22' AND email IS NOT NULL GROUP BY username, phone_number, email;

7) Seventh(2000 va 2024 yillar orasidan tug'ilgan odamlarni username, birthday va country sini olib bering):

SELECT username, birthday, country FROM users WHERE birthday BETWEEN '2020-01-01' AND '2024-08-27';

8) Eighth(Eng yoshi katta 10 ta odamni olib bering):

SELECT * FROM users ORDER BY birthday ASC LIMIT 10;

9) Ninth(Yoshi bo'yicha odamlarni tartiblang va 30 dan 40 gacha o'rinda turgan odamlarni olib bering):

SELECT * FROM users ORDER BY birthday ASC OFFSET 30 LIMIT 10;

10) Tenth(Odamlarni davlar bo'yicha guruhlab oling va faqatgina 10 tadan ko'p odam bo'lgan davlatlarni nomi va sonini olib bering):

SELECT country, COUNT(*) FROM users GROUP BY country HAVING COUNT(*) > 10;
